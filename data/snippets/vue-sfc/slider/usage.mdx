<script setup>
import * as slider from "@zag-js/slider";
import { normalizeProps, useMachine } from "@zag-js/vue";
import { computed } from "vue";

const [state, send] = useMachine(slider.machine({ id: "slider" }));
;
const api = computed(() => slider.connect(state.value, send, normalizeProps));
</script>

<template>
  <div ref="ref" v-bind="api.rootProps">
    <div>
      <label v-bind="api.labelProps">Slider Label</label>
      <output v-bind="api.outputProps">{{ api.value }}</output>
    </div>
    <div v-bind="api.controlProps">
      <div v-bind="api.trackProps">
        <div v-bind="api.rangeProps" />
      </div>
      <div v-bind="api.thumbProps">
        <input v-bind="api.inputProps" />
      </div>
    </div>
  </div>
</template>
